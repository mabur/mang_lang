num_fishes@{
    fishes = [3 4 3 1 2]
    simulate = in input out output@{
        output = []
        for fish in input
            output =
                if fish then
                    put!(dec!fish output)
                else
                    put_each!([6 8] output)
        end
    }
    days = 80
    for _ in days
        fishes = simulate!fishes
    end
    num_fishes = count!fishes
}
